SELECT S.CategoryName FROM db_book B, db_subject S, db_supplier S1 WHERE B.SupplierID = S1.SupplierID AND B.SubjectID = S.SubjectID AND S1.CompanyName = 'supplier2';
SELECT B.Title, B.UnitPrice From db_book B, db_supplier S1 Where B.SupplierID = S1.SupplierID AND B.UnitPrice = (select MAX(B1.UnitPrice) From db_book B1, db_supplier S2 Where B1.SupplierID = S2.SupplierID AND B1.UnitPrice AND S2.CompanyName = "Supplier3");
SELECT Distinct B.Title FROM db_book B, db_order O, db_order_detail OD, db_customer C WHERE B.BookID = OD.BookID AND OD.OrderID = O.OrderID AND O.CustomerID = C.CustomerID AND C.LastName = 'lastname1' AND C.FirstName = 'firstname1';
SELECT B.Title FROM db_book B WHERE B.Quantity > 10;
SELECT SUM(OD.Quantity*B.UnitPrice) FROM db_book B, db_order_detail OD, db_order O, db_customer C WHERE OD.BookID = B.BookID AND O.OrderID = OD.OrderID AND C.CustomerID = O.CustomerID AND C.FirstName = "firstname1" AND C.LastName="lastname1";
SELECT FirstName, LastName from (SELECT C.FirstName as FirstName, C.LastName as LastName, SUM(OD.Quantity*B.UnitPrice) AS PurchaseTotal FROM db_book B, db_order_detail OD, db_order O, db_customer C WHERE OD.BookID = B.BookID AND O.OrderID = OD.OrderID AND C.CustomerID = O.CustomerID GROUP BY C.CustomerID HAVING PurchaseTotal < 80) as temporary;
SELECT B.Title FROM db_book B, db_supplier S1 WHERE B.SupplierID = S1.SupplierID AND S1.CompanyName = 'supplier2';
SELECT C.FirstName, C.LastName, SUM(OD.Quantity*B.UnitPrice) AS PurchaseTotal FROM db_book B, db_order_detail OD, db_order O, db_customer C  WHERE OD.BookID = B.BookID AND O.OrderID = OD.OrderID AND C.CustomerID = O.CustomerID GROUP BY C.CustomerID ORDER BY PurchaseTotal DESC
SELECT B.Title, Shp.ShipperName FROM db_book B, db_shipper Shp, db_order O, db_order_detail OD WHERE B.BookID = OD.BookID AND OD.OrderID = O.OrderID AND O.ShipperID = Shp.ShipperID AND O.ShippedDate = "2016-08-04";
SELECT Distinct B.Title FROM db_book B, db_order O, db_order_detail OD, db_customer C WHERE B.BookID = OD.BookID AND OD.OrderID = O.OrderID AND O.CustomerID = C.CustomerID AND C.FirstName='firstname1' AND C.LastName='lastname1' AND B.BookID in (SELECT B1.bookID FROM db_book B1, db_order O1, db_order_detail OD1, db_customer C1 WHERE B1.BookID = OD1.BookID AND OD1.OrderID = O1.OrderID AND O1.CustomerID = C1.CustomerID AND C1.FirstName='firstname4' AND C1.LastName='lastname4');
SELECT B.Title FROM db_book B, db_employee E, db_order O, db_order_detail OD WHERE OD.BookID = B.BookID AND OD.OrderID = O.OrderID AND O.EmployeeID = E.EmployeeID AND E.FirstName = 'firstname6' AND LastName='lastname6';
SELECT B.Title, OD.Quantity FROM db_book B, db_order_detail OD WHERE B.BookID = OD.BookID ORDER BY OD.Quantity ASC;
SELECT FirstName, LastName from (select C.FirstName as FirstName, C.LastName as LastName, SUM(OD.Quantity) AS TotalBooksOrdered FROM db_customer C, db_order O, db_order_detail OD WHERE OD.OrderID = O.orderID AND O.CustomerID = C.CustomerID GROUP BY C.CustomerID HAVING TotalBooksOrdered >= 2) as Temp;
SELECT C.FirstName, C.LastName, B.Title FROM db_customer C, db_subject S, db_book B, db_order O, db_order_detail OD WHERE O.OrderID = OD.OrderID AND B.BookID = OD.BookID AND O.CustomerID = C.CustomerID AND B.SubjectID = S.SubjectID AND (S.CategoryName = 'category3' OR S.CategoryName = 'category4');
SELECT C.FirstName, C.LastName FROM db_order O, db_order_detail OD, db_customer C, db_book B WHERE O.OrderID = OD.OrderID AND O.CustomerID = C.CustomerID AND OD.BookID = B.BookID AND B.Author = 'author1';
SELECT E.FirstName, E.LastName, SUM(B.UnitPrice * OD.Quantity) AS OrderPrice FROM db_book B, db_employee E, db_order O, db_order_detail OD WHERE O.OrderID = OD.OrderID AND B.BookID = OD.BookID AND O.EmployeeID = E.EmployeeID GROUP BY E.EmployeeID;
SELECT B.Title, sum(OD.Quantity) as Total FROM db_book B, db_order O, db_order_detail OD  WHERE B.BookID = OD.BookID AND O.OrderID = OD.OrderID AND (O.ShippedDate is null or O.ShippedDate > '2014-08-04') group by B.Title;
SELECT C.FirstName, C.LastName, SUM(OD.Quantity) AS TotalBooksOrdered FROM db_customer C, db_order O, db_order_detail OD WHERE OD.OrderID = O.orderID AND O.CustomerID = C.CustomerID GROUP BY C.CustomerID HAVING TotalBooksOrdered > 1 ORDER BY TotalBooksOrdered DESC;
SELECT C.FirstName, C.LastName, C.Phone, SUM(OD.Quantity) AS TotalBooksOrdered FROM db_customer C, db_order O, db_order_detail OD WHERE OD.OrderID = O.orderID AND O.CustomerID = C.CustomerID GROUP BY C.CustomerID HAVING TotalBooksOrdered >3;
